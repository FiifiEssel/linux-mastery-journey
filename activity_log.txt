#User & Group Management
#adding users neo, trinity, morpheus

sudo adduser neo

sudo adduser morpheus

# adding groups
sudo groupadd redteam

#adding users to groups
#neo and trinity to redteam

sudo usermod -aG redteam neo
sudo usermod -aG redteam trinity

sudo usermod -aG analysts morpheus

#granting morpheus sudo rights
sudo usermod -aG sudo morpheus
