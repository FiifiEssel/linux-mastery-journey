#INITIAL SET-UP

sudo mkdir /srv/cyberlab

#creating a README.md
sudo touch /srv/cyberlab/README.md

#creating activity_log.txt
sudo touch /srv/cyberlab/activity_log.txt

#USER & GROUP MANAGEMENT
#adding users neo, trinity, morpheus

sudo adduser neo
sudo adduser trinity
sudo adduser morpheus

# adding groups
sudo groupadd redteam
sudo groupadd analysts

#adding users to groups
#neo and trinity to redteam

sudo usermod -aG redteam neo
sudo usermod -aG redteam trinity

sudo usermod -aG analysts morpheus

#granting morpheus sudo rights
sudo usermod -aG sudo morpheus


#DIRECTORY AND FILE PERMISSIONS
#creating a directory in the /srv/redteam-data

sudo mkdir /srv/redteam-data

#setting ownership to redteam
sudo chown :redteam /srv/redteam-data

#applying permissions for shared collaboration
sudo chmod 2770 redteam-data

#creating a file /srv/cyberlab/brief.txt
#setting ownership to morpheus and group analysts
sudo chown :morpheus /srv/cyberlab/brief.txt
sudo chown :analysts /srv/cyberlab/brief.txt

#adding permssions to only morpheus and analysts
sudo chmod 2770 /srv/cyberlab/brief.txt
